import (
	"fmt"
	"okey-dokey/ok_action"
	"okey-dokey/bad"
)

func defaultLessThanOrEqualFormat(definition *LessThanOrEqual, value *{{.PrimitiveKeyword}}) string {
	return fmt.Sprintf("must be less than or equal to %d", definition.Value)
}

type LessThanOrEqual struct {
	Format func(definition *LessThanOrEqual, value *{{.PrimitiveKeyword}}) string
	Value  {{.PrimitiveKeyword}}
}

func (m *LessThanOrEqual) Validate(value *{{.PrimitiveKeyword}}, violationReceiver bad.MessageReceiver) ok_action.Enum {
	formatter := defaultLessThanOrEqualFormat
	if m.Format != nil {
		formatter = m.Format
	}
	if value == nil {
		return ok_action.Continue
	}
	if *value > m.Value {
		violationReceiver.ReceiveMessage(formatter(m, value))
	}
	return ok_action.Continue
}
