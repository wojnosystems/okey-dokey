import (
	"fmt"
	"okey-dokey/ok_action"
	"okey-dokey/bad"
)

func defaultGreaterThanFormat(definition *GreaterThan, value *{{.PrimitiveKeyword}}) string {
	return fmt.Sprintf("must be greater than %d", definition.Value)
}

type GreaterThan struct {
	Format func(definition *GreaterThan, value *{{.PrimitiveKeyword}}) string
	Value  {{.PrimitiveKeyword}}
}

func (m *GreaterThan) Validate(value *{{.PrimitiveKeyword}}, violationReceiver bad.MessageReceiver) ok_action.Enum {
	formatter := defaultGreaterThanFormat
	if m.Format != nil {
		formatter = m.Format
	}
	if value == nil {
		return ok_action.Continue
	}
	if *value <= m.Value {
		violationReceiver.ReceiveMessage(formatter(m, value))
	}
	return ok_action.Continue
}
